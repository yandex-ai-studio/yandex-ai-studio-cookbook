package
main

import
(

"context"

"encoding/json"

"fmt"

"log"

"github.com/openai/openai-go"

"github.com/openai/openai-go/option"

"github.com/openai/openai-go/responses"

)

const
(
    YANDEX_MODEL     =
"yandexgpt"

    YANDEX_API_KEY   =
""

    YANDEX_FOLDER_ID =
""

)

func
main
()
{
    client := openai.NewClient(
        option.WithAPIKey(YANDEX_API_KEY),
        option.WithBaseURL(
"https://ai.api.cloud.yandex.net/v1"
),
        option.WithProject(YANDEX_FOLDER_ID),
    )

    model := fmt.Sprintf(
"gpt://%s/%s"
, YANDEX_FOLDER_ID, YANDEX_MODEL)

    response, err := client.Responses.New(context.Background(), responses.ResponseNewParams{
        Model: model,
        Input: responses.ResponseNewParamsInputUnion{
            OfString: openai.String(
"Сделай краткий обзор новостей про LLM за сентябрь 2025 года."
),
        },
        Tools: []responses.ToolUnionParam{
            {
                OfWebSearchPreview: &responses.WebSearchToolParam{
                    Type:
"web_search_preview"
,
                },
            },
        },
        Temperature:     openai.Float(
0.3
),
        MaxOutputTokens: openai.Int(
1000
),
    })

if
err !=
nil
{
        log.Fatalf(
"Ошибка при создании ответа: %v"
, err)
    }

// Текст ответа

    fmt.Println(
"Текст ответа:"
)
    fmt.Println(response.OutputText())
    fmt.Println(
"\
"
+
"==================================================\
"
)

// Полный ответ

    fmt.Println(
"Полный ответ (JSON):"
)
    jsonBytes, err := json.MarshalIndent(response,
""
,
"  "
)

if
err !=
nil
{
        log.Fatalf(
"Ошибка сериализации JSON: %v"
, err)
    }
    fmt.Println(
string
(jsonBytes))
}
